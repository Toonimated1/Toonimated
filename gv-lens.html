<script>

/* Tunables via URL: ?fps=12&bands=28&slit=18&jump=18&amp=32&alpha=0.9

   (All optional; sensible defaults below.) */

(function(){

  const q=new URLSearchParams(location.search);

  const FPS=+q.get('fps')||12, BANDS=+q.get('bands')||28, SLIT=+q.get('slit')||18;

  const JUMP=+q.get('jump')||18, AMP=+q.get('amp')||32, ALPHA=+q.get('alpha')||0.9;

  const L=document.getElementById('L'), H=()=>innerHeight;

  for(let i=0;i<BANDS;i++){const d=document.createElement('div');d.className='b';d.style.height=SLIT+'px';d.style.opacity=ALPHA;L.appendChild(d)}

  const bands=[...L.children], Y=bands.map(()=>Math.random()*H());

  addEventListener('resize',()=>{/* nothing; H() reads live */});

  let last=0, step=1000/FPS;

  function frame(t){

    if(t-last>=step){ last=t;

      const h=H();

      for(let i=0;i<bands.length;i++){

        Y[i]=(Y[i]+JUMP+Math.random()*JUMP*0.2)%h;

        const shift=(Math.random()*2-1)*AMP;

        const b=bands[i];

        b.style.top=Y[i]+'px';

        b.style.transform=`translate3d(${shift}px,0,0)`;

        b.style.background=`linear-gradient(90deg,rgba(255,255,255,${0.05+Math.random()*0.08}),rgba(255,255,255,.95),rgba(255,255,255,${0.05+Math.random()*0.08}))`;

      }

    }

    requestAnimationFrame(frame);

  }

  requestAnimationFrame(frame);

})();

</script>
